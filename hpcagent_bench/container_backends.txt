# Container backend spellings -- the SINGLE source of truth for how each runtime
# spells its launch flags. Read by BOTH hpcagent_bench/containers.py (Python) and
# scripts/run_agent_in_container.sh (pure bash, for a host with no Python), so the
# two launch paths cannot drift. Format: `<backend>.<field>=<value>`, one per line
# (a .properties/.env shape -- deliberately NOT JSON/YAML so bash can read it with a
# three-line `while IFS='=' read` loop). Blank lines and `#` comments are ignored.
#
# Only apptainer + podman are supported: both are what CSCS launches, and both consume
# the ONE universal OCI image (apptainer builds a SIF from it; podman runs the OCI tag
# directly). A multi-node inference endpoint (a vLLM behind >1 node) is a ray cluster of
# these single-node containers exposing ONE URL -- no container ever spans nodes.
#
# INVARIANT: every value is a whitespace-separated TOKEN LIST with no intra-token
# spaces (`run --rm --network host` is three tokens; a path/env value with a space
# would break both splitters -- those are computed at runtime, never stored here).
# Env ordering is pinned identically on both sides (HPCAGENT_BENCH_IMAGE first, then this
# passthrough list in order, then the remaining HPCAGENT_BENCH_* vars sorted under LC_ALL=C
# == Python's default str sort) so the assembled argv is byte-identical.

global.passthrough=HPCAGENT_BENCH_OLLAMA_HOST OLLAMA_HOST HPCAGENT_BENCH_OLLAMA_MODEL ANTHROPIC_API_KEY HPCAGENT_BENCH_LOCAL_MODEL

apptainer.verb=exec
apptainer.bind=--bind
apptainer.workdir=--pwd
apptainer.env=--env
apptainer.gpu.nvidia=--nv
apptainer.gpu.amd=--rocm
apptainer.image_form=sif
apptainer.image_default=hpcagent_bench-{hw}.sif
apptainer.harbor_env=singularity

podman.verb=run --rm --network host
podman.bind=-v
podman.workdir=-w
podman.env=-e
podman.gpu.nvidia=--device nvidia.com/gpu=all
podman.gpu.amd=--device /dev/kfd --device /dev/dri --group-add keep-groups
podman.image_form=tag
podman.image_default=hpcagent_bench:{hw}
podman.harbor_env=
